  // Workspace invitation management
  async rejectWorkspaceInvite(workspaceId: string): Promise<void> {
    await this.post(`/workspaces/${workspaceId}/reject-invite`, {});
  }

  async getSentInvitations(workspaceId: string): Promise<any[]> {
    const response = await this.get(`/workspaces/${workspaceId}/invitations`);
    return response.data || [];
  }

  async getReceivedInvitations(): Promise<any[]> {
    const response = await this.get('/workspaces/invitations/received');
    return response.data || [];
  }

  async cancelWorkspaceInvite(workspaceId: string, invitationId: string): Promise<void> {
    await this.delete(`/workspaces/${workspaceId}/invitations/${invitationId}`);
  }

  // Join request management
  async cancelJoinRequest(workspaceId: string): Promise<void> {
    await this.delete(`/workspaces/${workspaceId}/join-request`);
  }
