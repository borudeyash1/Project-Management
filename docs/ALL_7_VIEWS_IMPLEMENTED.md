```typescript

  ? myTasks 
  : myTasks.filter(task => task.status === filterStatus);

// Each view uses filteredTasks
{viewMode === 'list' && filteredTasks.map(...)}
{viewMode === 'kanban' && filteredTasks.filter(...).map(...)}
{viewMode === 'calendar' && filteredTasks.filter(...).map(...)}
```
